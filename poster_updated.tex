% ====================================================== COLUMN 1
\begin{block}{Overview}
The CMS pixel detector (1{,}856 modules) is the innermost tracker sub-system. Through Run~3
the LHC has delivered $>$350~fb$^{-1}$, causing significant \textbf{radiation
damage} to the barrel pixel layer closest to the beam (BPIX~L1). The degradation of charge
collection efficiency is most pronounced at large pseudorapidity, where shallow track
angles --- due to the per-pixel charge threshold --- can lead to
\textbf{truncated or broken clusters} that bias hit reconstruction.
\par\smallskip
\emph{Digi morphing} is a mitigation that applies a \textbf{morphological closing}
at the pixel-\emph{digi} level to recover cluster connectivity. It was deployed
in offline and HLT reconstruction during 2025.
\end{block}

\bsep
\begin{block}{Motivation: broken \& shortened clusters}
Radiation damage lowers the charge collection efficiency, so charge deposits can fall
below the per-pixel threshold. For \textbf{shallow-angle} tracks at high $|\eta|$ this
leaves:
\begin{itemize}
  \item \textbf{broken clusters}: internal gaps split one cluster into fragments;
  \item \textbf{shortened clusters}: the reconstructed extent is truncated.
\end{itemize}
Digi morphing recovers \textbf{broken} clusters by bridging internal gaps, but
\textbf{cannot recover shortened clusters}, where charge is lost at the cluster ends.
\stackfig{posterfigs/sensor_broken_hr.png}{posterfigs/sensor_short_hr.png}{0.85}{Sensor
cross-sections: along a shallow track the collected charge can dip below the
per-pixel threshold, \textbf{breaking} the cluster into fragments (top) or
\textbf{shortening} its reconstructed extent (bottom). Adapted from
Chiochia~et~al.~[1].}
\end{block}

\bsep
\begin{block}{The digi morphing algorithm}
An image-processing \textbf{morphological closing} is applied directly to the
pixel digis:
\begin{itemize}
  \item \textbf{Dilation:} a $3{\times}3$ kernel expands cluster boundaries,
        bridging the internal sub-threshold gaps;
  \item \textbf{Erosion:} restores the original cluster extent while keeping the
        recovered pixels.
\end{itemize}
The inserted \textbf{helper digis} are temporary connectivity markers --- they
carry no charge and are \emph{not} used to estimate the hit position, so the
position remains unbiased.
\posterfig{posterfigs/cand_E_final_T.pdf}{0.92}{\textbf{Figure~1:} BPIX~L1 module
($\eta\!\sim\!-2.3$, 25~July~2024). Helper digis (black) added in dilation, then
eroded, let the clusterizer merge digis from a single broken cluster.}
\end{block}

\bsep
\begin{block}{Implementation \& deployment}
\begin{itemize}
  \item Written in the \textbf{alpaka} portability library~[2] $\Rightarrow$ runs
        unchanged on \textbf{CPU and GPU}; inserted before the pixel clusterizer in
        pixel local reconstruction at HLT.
  \item Morphed regions are \textbf{fully configurable} via configuration flags.
  \item Merged into \textbf{CMSSW}, featuring several subsequent optimizations.
  \item \textbf{Deployed at the HLT} from LHC Fill~11200 (17~October~2025).
\end{itemize}
\end{block}

\bsep
\begin{block}{Impact on clusters in data}
\posterfig{posterfigs/fig3_303042568.pdf}{0.94}{\textbf{Figure~2:} cluster
reconstruction with morphing disabled (left, 12 clusters) and enabled (right, 5
clusters), BPIX~L1, $\eta\!\sim\!-2.4$, 14~October~2025. Broken fragments are
reconnected and nearby components merged, reducing combinatorics.}
Beyond reconnecting broken clusters, morphing \textbf{merges fragments consistent
with the same track incidence angle}, lowering the number of reconstructed clusters.
Limitations are \textbf{over-merging} of fragments from different tracks and
\textbf{under-merging} when the gap is too large for the kernel to bridge.
\end{block}

\begin{block}{Application: targeted regions}
Morphing is enabled only in the BPIX module rings (along $z$) that see the most
shallow-angle tracks and the largest cluster-size loss:
\begin{itemize}
  \item \textbf{BPIX Layer~1:} rings 1, 2, 7, 8 \;($|\eta|\gtrsim2.2$);
  \item \textbf{BPIX Layer~2:} rings 1, 8 \;($|\eta|\gtrsim1.8$).
\end{itemize}
This conservative, region-targeted choice confines the algorithm to the phase
space where it helps most, leaving the rest of the detector untouched.
\pairfig{posterfigs/config_grid_hr.png}{posterfigs/detector_regions_hr.png}{0.49}{Enabled
(\textbullet) vs disabled ($\otimes$) module rings (left), and the corresponding
highlighted regions on the Phase-1 pixel detector (right, modified from~[3]).}
\end{block}

\bsep
\begin{block}{HLT local hit resolution}
\textbf{Simulated single muons} with no additional collisions (no pile-up), each
with $p_{T}\approx10$~GeV and crossing the forward part of the inner barrel layer
($2<|\eta|<3$, BPIX~L1). The reconstructed hit is compared to the true simulated
position. The residual in local~$y$ is \textbf{bimodal}: broken or shortened
clusters pull the hit in opposite directions depending on the track's incidence
angle.
\pairfig{posterfigs/recHitXResLayer1.pdf}{posterfigs/recHitYResLayer1.pdf}{0.45}{\textbf{Figure~3:}
hit residual (RecHit $-$ SimHit) in local~$x$ (left) and local~$y$ (right), BPIX~L1.
Blue: morphing disabled; red: enabled.}
\begin{itemize}
  \item Local-$y$ RMS \textbf{256.6\,$\rightarrow$\,213.1\,$\mu$m} --- the bimodal
        lobes collapse toward zero;
  \item Local-$x$ essentially unchanged (33.0\,$\rightarrow$\,32.4\,$\mu$m) since
        radiation damage resulting in broken clusters mainly affects the local-$y$
        coordinate.
\end{itemize}
The no-pile-up configuration isolates the local-reconstruction effect, so the RMS
reduction directly measures the recovered hit quality.
\end{block}

\bsep
\begin{block}{HLT tracking: efficiency \& purity}
\textbf{Simulated $\mathrm{t\bar{t}}$ events} at $\sqrt{s}=13.6$~TeV with realistic
Run-3 pile-up.
\pairfig{posterfigs/effic.pdf}{posterfigs/fakeduprate_vs_eta.pdf}{0.435}{\textbf{Figure~4:}
track-finding efficiency (left) and fake$+$duplicate rate (right) vs track $\eta$.
Gains are concentrated at high $|\eta|$.}
\begin{itemize}
  \item Efficiency up \textbf{$\sim$5\% (15\%)} for $2.0<|\eta|<2.5$
        ($|\eta|>2.5$);
  \item simultaneous reduction of the fake$+$duplicate rate.
\end{itemize}
The gain mirrors the cluster recovery seen in data and is concentrated in the
forward region, where broken clusters are most frequent --- the central region lies
outside the morphed rings and is unchanged by construction.
\end{block}

\bsep
\begin{block}{HLT tracking: impact parameter \& $\boldsymbol{p_{T}}$}
\pairfig{posterfigs/dzres_vs_eta_Sigma.pdf}{posterfigs/ptres_vs_eta_Sigma.pdf}{0.46}{\textbf{Figure~5:}
resolution of the longitudinal impact parameter $d_{z}$ (left) and of $p_{T}$
(right) vs track $\eta$.}
\begin{itemize}
  \item $d_{z}$ resolution improved by up to \textbf{$\sim$50\%} at high $|\eta|$;
  \item $p_{T}$ resolution unchanged
\end{itemize}
\textbf{Better hits $\Rightarrow$ better tracks}: the recovered local-$y$
resolution feeds directly into improved efficiency and impact-parameter
resolution in the forward region. The $d_{z}$ gain is especially relevant for
primary-vertex association and pile-up rejection, while the unchanged $p_{T}$
resolution confirms that morphing adds no bias to the track curvature
measurement. All metrics are evaluated with the standard Run-3 HLT tracking
validation~[4].
\end{block}

\begin{block}{Cluster-size trends in data}
The most-irradiated \textbf{BPIX~L1} cluster size, which had fallen fill by fill,
\textbf{stabilises} once digi morphing is enabled --- in \emph{both} transverse
directions and most prominently for the outer ladders.

\pairfig{posterfigs/hlt_sizey_vs_eta_inner_0.pdf}{posterfigs/hlt_sizey_vs_eta_outer_0.pdf}{0.498}{\textbf{Figure~7

split in pseudorapidity, for inner (left) and outer (right) BPIX~L1 ladders.
Dashed line: 17~October~2025, where morphing was enabled at the HLT.}
\end{block}

\bsep
\begin{block}{Physics performance: $\boldsymbol{J/\psi}$ scouting}
The HLT scouting stream is taken at a higher rate with reduced event content and no
offline reconstruction~[7]. The downstream physics impact is measured with
HLT-scouting dimuons without vertex constraint (NoVtx), comparing data
\textbf{before} and \textbf{after} morphing was enabled at the HLT on 17~October~2025.
\pairfig{posterfigs/jpsi_n_sig_vs_eta.png}{posterfigs/fit_eta_11.png}{0.495}{\textbf{Figures~8--9:}
J/$\psi$ peak yield vs dimuon $\eta_{\mu\mu}$ (left, lower panel: relative shift),
and the dimuon invariant-mass peak for $2.0\le\eta_{\mu\mu}<2.4$ (right).}
\begin{itemize}
  \item J/$\psi$ yield rises by \textbf{up to $\sim$8--9\%} in the forward region,
        where radiation damage is worst;
  \item accompanied by a \textbf{narrower, taller J/$\psi$ mass peak} --- a direct
        sign of improved forward tracking.
\end{itemize}
\textbf{Better tracks $\Rightarrow$ better physics.}
\end{block}

\bsep
\begin{block}{Timing \& throughput}
20{,}000 events on HLT nodes (2$\times$AMD~EPYC~7763 $+$ 2$\times$NVIDIA~T4), 12
timed repetitions after warm-up.
\par\vspace{2mm}
{\centering\renewcommand{\arraystretch}{1.18}%
\begin{tabular}{@{}p{0.40\blockin}>{\raggedleft\arraybackslash}p{0.27\blockin}>{\raggedleft\arraybackslash}p{0.27\blockin}@{}}
\toprule
\textbf{Configuration} & \textbf{Time/event (ms)} & \textbf{Events/s}\\
\midrule
Morphing \textcolor{cmsred}{OFF} & $480.0\pm0.5$ & $538.9\pm1.7$\\
Morphing \textcolor{cmsgreen}{ON}  & $482.0\pm0.7$ & $536.5\pm2.4$\\
\bottomrule
\end{tabular}\par}
\vspace{2mm}
The Pixel ($\sim$10.4\%) and Tracking ($\sim$23\%) HLT time fractions are
unchanged $\Rightarrow$ digi morphing comes at a \textbf{negligible cost}. The
morphological closing is a small, fully parallel operation per module, so it adds
no measurable latency to the GPU pixel-reconstruction step and leaves the overall
HLT throughput budget intact.
\end{block}

Intervention at \emph{local reconstruction}, motivated by detector physics,
propagates all the way to physics observables:
